    def simple_tokennize(s:str):
      return re.findall(r"\b\w+\b", s.lower())

    MAX_VOCAB = 10000
    specials = ["<unk>", "<pad>"]

    counter = Counter()
    for _, txt in train_list:
      counter.update(simple_tokennize(txt))

    most_common = [w for w, _ in counter.most_common(MAX_VOCAB - len(specials))]
    itos = specials + most_common
    stoi = {w: i for i,w in enumerate(itos)}
    UNK_IDX = stoi["<unk>"]
    PAD_IDX = stoi["<pad>"]

    def collate_batch(batch):
      text_list, label_list = [], []
      for (label, text) in batch:
      text_list.append(text_pipeline(text))
      label_list.append(label_pipeline(label))
  
      text_padded = pad_sequence(text_list, batch_first=True, padding_value = PAD_IDX)
      labels = torch.stack(label_list)
      return text_padded.to(DEVICE), labels.to(DEVICE)

    class SentimentMLP(nn.Module):
    
    def __init__(self, vocab_size, embed_dim=64, hidden_dims=(32,32), pad_idx=None):
    
    super().__init__()
    
    self.embedding=nn.Embedding(vocab_size, embed_dim, padding_idx=pad_idx if pad_idx is not None else 0)
    
    layers = []
    
    in_dim = embed_dim
    
    for h in hidden_dims:
    
      layers += [nn.Linear(in_dim, h), nn.ReLU()]
      in_dim = h
      
    layers += [nn.Linear(in_dim,1)]
    self.mlp = nn.Sequential(*layers)

- class ~~~ -> 새로운 모델 클래스, nn.Module 상속
- def __init__() -> 클래스가 만들어질 때 자동으로 실행되는 초기화 함수
- > self -> 만들어지는 모델 자기자신 --> self.embedding 모델 내부 부품을 저장할 때 사용
  >
  > vocab_size = > 단어사전 크기 -> 임베딩 층에서 몇개의 단어를 처리할지 정하는 값
  >
  > embed_dim => 각 단어를 몇 차원 벡터로 바꿀지 기본값 64 --> 단어하나를 길이 64짜리 벡터로 변환
  >
  > hidden_dims = (32,32) --> 은닉층 크기들을 튜플로 넣은 것
  >
  > pad_idx = None --> padding 토큰 번호
  >

- self.embedding=nn.Embedding(vocab_size, embed_dim, padding_idx=pad_idx if pad_idx is not None else 0)
- > 단어 번호를 64차원 벡터로 바꾸는 Embedding 층을 만들고, PAD 토큰은 따로 지정하겠다
  
- layers -> MLP를 구성할 층들을 담아둘 빈 리스트 -> 나중에 리니어, 렐루 넣을꺼

### - forward 함수 만들기

    def forward(self, x):
      emb = self.embedding(x)

      if hasattr(self.embedding, "padding_idx") and self.embedding.padding_idx is not None:
        pad_idx = self.embedding.padding_idx
      else:
        pad_idx =0
  
      mask = (x!=pad_idx).unsqueeze(-1)
      
      emb = emb * mask
      
      lengths = mask.sum(dim=1).clarm(min=1)
      
      mean_pooled = emb.sum(dim=1)/lengths
      
      logits = self.mlp(mean_pooled).squeeze(1)
      
      return logits


 ###  - 이 함수는 문장 안 단어들을 벡터로 바꾼뒤, 패딩을 빼고 진짜 단어들만 평균 내서 문장이 긍정인지 부정인지 점수 하나 뽑는과정

 - x 는 단어 번호표들의 배열
 - > shape -> (batch, T ) * T 는 맞춰진 길이

- emb = self.embedding(x) -> 각 숫자 단어를 숫자벡터로 바꿈 -> 단어 하나를 길이 E짜리 벡터로 바꿈
- > emb.shape --> (B, T , E)
  >
- 평균낼껀데 pad 값 있는 건 안됨
- pad 값 뺄꺼임

- if ~~~ -> 어느 숫자가 PAD인지 알아내는 관계

-  mask = (x!=pad_idx).unsqueeze(-1) -> pad = 0 -> PAD가 아닌 곳찾기
-  unsqueeze(-1) -> 차원 하나 늘려주기 -> 앞에 임베딩이랑 곱하기위함

-  emb = emb * mask -> 임베딩에서 pad값 0으로 만들기

-  mask.sum(dim=1) -> 문장마다 더하기 [1,1,1,1,0] -> [4] -> shape(B,1)
-  mean_pooled = emb.sum(dim=1) / lengths
-  > 임베딩 단어 백터 다 더하기
